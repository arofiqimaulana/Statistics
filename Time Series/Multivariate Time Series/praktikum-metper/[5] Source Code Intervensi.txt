*Import data;
proc import out=work.Intervensi datafile='D:\BANK DATA\DATABASE\INTERVENSI\IKK.xlsx' 
dbms=xlsx replace;
sheet='sheet1';
getnames=yes;
run;
*Print data;
proc print data=Intervensi;
run;
*plot data time series;
proc gplot data=Intervensi;
symbol1 i=splines v=dot;
plot IKK*Date ;
run;
*manipulasi data;
data Intervensi;
set Intervensi;
z=0;
tx=IKK**0.75;
run;
*Stasioneritas ragam;
*transformasi box cox;
proc transreg maxiter=0 nozeroconstant;
      model BoxCox(IKK) = identity(z);
      output;
run;
*Stasioneritas rata-rata;
proc arima data=Intervensi;
identify var=tx nlag=40 stationarity=(adf);
run;
proc arima data=Intervensi;
run;
*Pembentukan model ARIMA dan identifikasi (b,r,s);
identify var=tx(1) nlags=25;
run;
estimate p=(0) q=(3) noconstant method=ml;
run;
forecast out=ramalan lead=12 printall;
run;
*Pembentukan model ARIMA+Intervensi;
proc arima data=Intervensi;
identify var=tx(1) crosscorr=(news(1));
estimate p=(0) q=(3) noconstant method=ml;
run;
estimate p=(0) q=(3) input=(9 $(4)/(0)news) noconstant method=ml;
forecast out=ramalan2 lead=5 ;
run;
proc univariate normaltest;
var residual;
run;
